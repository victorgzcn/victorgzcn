import win32com.client as win32

def send_outlook_email(recipient, subject, body):
    outlook = win32.Dispatch('Outlook.Application')
    mail = outlook.CreateItem(0)  # 0 represents MailItem
    mail.To = recipient
    mail.Subject = subject
    mail.Body = body
    mail.Send()

# Example usage:
send_outlook_email(
    recipient="victor@plyflame.com",
    subject="Test Email from Python",
    body="This is a test email sent via Outlook using Python."
)
